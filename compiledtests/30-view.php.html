Add/Alter/Drop a view
Login as janus
open http://localhost:8800/ppa/intro.php
select name=language English
open http://localhost:8800/ppa/login.php?server=localhost:5432:allow&subject=server
type //input[@name='loginUsername'] admin_user
type //input[@id='loginPassword'] super
clickAndWait name=loginSubmit
assertText //div[@class='topbar']/descendant::span[@class='username'] admin_user
1. Create the view
clickAndWait link=Databases
clickAndWait link=ppatests_db
clickAndWait link=Schemas
clickAndWait link=public
clickAndWait link=Views
clickAndWait link=Create view
type name=formView student_promo
type name=formDefinition SELECT s.id, name, birthday, resume, spe, year FROM student AS s JOIN test_schema.promo AS p ON (s.id_promo=p.id)
type name=formComment students and their promotion
clickAndWait //input[@value='Create']
assertText //p[@class='message'] View created.
assertText //tr/td[2]/a[text()='student_promo']/../../td[2] student_promo
assertText //tr/td[2]/a[text()='student_promo']/../../td[3] admin_user
assertText //tr/td[2]/a[text()='student_promo']/../../td[8] students and their promotion
assertText //div[@class='trail']/descendant::a[@title='Schema']/span[@class='label'] public
2. Alter the view
clickAndWait link=Views
clickAndWait link=student_promo
clickAndWait //ul[@class='navlink']/li/a[text()='Alter']
2.1. Alter view's name
type name=name student_promo_renamed
clickAndWait name=alter
assertText //p[@class='message'] View altered.
assertText //div[@class='trail']/descendant::a[@title='View']/span[@class='label'] student_promo_renamed
2.2. Alter view's comment
clickAndWait //ul[@class='navlink']/li/a[text()='Alter']
type name=comment students and their promotion (altered)
clickAndWait name=alter
assertText //p[@class='message'] View altered.
assertText //p[@class='comment'] students and their promotion (altered)
2.3. Alter view's schema
clickAndWait //ul[@class='navlink']/li/a[text()='Alter']
select name=newschema label=test_schema
clickAndWait name=alter
assertText //p[@class='message'] View altered.
assertText //div[@class='trail']/descendant::a[@title='Schema']/span[@class='label'] test_schema
2.4. Alter view's owner
clickAndWait //ul[@class='navlink']/li/a[text()='Alter']
select name=owner label=ppa_tests_user
clickAndWait name=alter
assertText //p[@class='message'] View altered.
clickAndWait //div[@class='trail']/descendant::a[@title='Schema']/span[@class='label' and text()='test_schema']
clickAndWait link=Views
assertText //tr/td[2]/a[text()='student_promo_renamed']/../../td[3] ppa_tests_user
2.5. Alter back everything
clickAndWait link=student_promo_renamed
clickAndWait //ul[@class='navlink']/li/a[text()='Alter']
select name=owner label=admin_user
type name=name student_promo
select name=newschema label=public
type name=comment students and their promotion
clickAndWait name=alter
assertText //p[@class='message'] View altered.
assertText //p[@class='comment'] students and their promotion
assertText //div[@class='trail']/descendant::a[@title='View']/span[@class='label'] student_promo
clickAndWait //div[@class='trail']/descendant::a[@title='Schema']/span[@class='label' and text()='public']
clickAndWait link=Views
assertText //tr/td[2]/a[text()='student_promo']/../../td[3] admin_user
3. Drop the view
clickAndWait //tr/td/a[text()='student_promo']/../../td/a[text()='Drop']
clickAndWait name=drop
assertText //p[@class='message'] View dropped.
assertErrorOnNext Element //tr/td/a[text()='student_promo'] not found
clickAndWait //tr/td/a[text()='student_promo']
Logout
clickAndWait //div[@class='trail']/descendant::tr/td[1]/a/span[@class='label' and text()='phpPgAdmin']
clickAndWait link=Servers
clickAndWait //tr/td/a[text()='PostgreSQL']/../../td/a[text()='Logout']
assertText //p[@class='message'] Logged out of PostgreSQL
